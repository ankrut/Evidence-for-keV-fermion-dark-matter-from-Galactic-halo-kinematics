% There is a bug of the bibliography style elsarticle-num-names.bst when the package hyperref is loaded. The fellowing code is a workaround found at https://www.tug.org/pipermail/tex-live/2012-August/032152.html

\makeatletter
\providecommand{\doi}[1]{%
  \begingroup
    \let\bibinfo\@secondoftwo
    \urlstyle{rm}%
    \href{http://dx.doi.org/#1}{%
      doi:\discretionary{}{}{}%
      \nolinkurl{#1}%
    }%
  \endgroup
}
\makeatother
